#### or的なこと
.or メソッドというのがあるんですねー。

    it "card is require boolean" do
    expect(@record.card).to match(true).or match(false)
    end

#### truthy/falsey
true と false で比較してくれる

    it "empty category value is invalid" do
    @record.category = nil
    expect(@record.save).to be_falsey
    end
