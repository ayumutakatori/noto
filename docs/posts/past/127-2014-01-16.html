def create

@kakeibos = Kakeibo.new(kakeibo_params)

@kakeibos.save

redirect_to kakeibos_path

end


private # <= よくわからないけど、セキュリティ的に必要だそうです。

def kakeibo_params

params[:kakeibo].permit(:day, :money, :category1, :category2, :card, :memo)

end
